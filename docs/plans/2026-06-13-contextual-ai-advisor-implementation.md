# Contextual AI Advisor Implementation Plan

> Steps use checkbox (`- [ ]`) syntax for progress tracking.

**Goal:** Build a scene-aware AI advisor for health, investment, AI, author, and article pages, backed by database-configurable RAG knowledge sources and PostgreSQL `pgvector` retrieval.

**Architecture:** Extend the existing FastAPI `ai_assistant` and `knowledge_base` contexts instead of creating a parallel AI stack. Keep public advisor orchestration in `ai_assistant`, knowledge source/index/vector concerns in `knowledge_base`, lightweight management in `admin-console`, and reusable advisor UI in `site`.

**Tech Stack:** FastAPI, SQLAlchemy 2.0, Alembic, PostgreSQL + `pgvector`, SQLite fallback, Pydantic, Next.js 15, React 19, TypeScript, Vitest, Vite.

from __future__ import annotations

from enum import StrEnum


class KnowledgeSourceKind(StrEnum):
    PUBLISHED_CONTENT = "published_content"
    MANUAL_NOTE = "manual_note"
    STRUCTURED_PROFILE = "structured_profile"
    EXTERNAL_LINK_PLACEHOLDER = "external_link_placeholder"


class KnowledgeSourceStatus(StrEnum):
    ENABLED = "enabled"
    DISABLED = "disabled"
    DELETED = "deleted"


KNOWN_DOMAINS = {"health", "investment", "ai", "author", "general"}


def normalize_domains(values: list[str]) -> list[str]:
    normalized: list[str] = []
    for value in values:
        item = value.strip().lower()
        if item in KNOWN_DOMAINS and item not in normalized:
            normalized.append(item)
    return normalized or ["general"]
```

class KnowledgeSourceDAO:
    def __init__(self, session: AsyncSession) -> None:
        self._session = session

    async def get_by_id(self, source_id: str) -> KnowledgeSourcePO | None:
        return await self._session.get(KnowledgeSourcePO, source_id)

    async def list_active(self) -> list[KnowledgeSourcePO]:
        result = await self._session.scalars(
            select(KnowledgeSourcePO).where(KnowledgeSourcePO.status != "deleted").order_by(KnowledgeSourcePO.updated_at.desc())
        )
        return list(result)

    async def add(self, source: KnowledgeSourcePO) -> KnowledgeSourcePO:
        self._session.add(source)
        await self._session.flush()
        await self._session.refresh(source)
        return source

    async def save(self, source: KnowledgeSourcePO) -> KnowledgeSourcePO:
        await self._session.flush()
        await self._session.refresh(source)
        return source

    async def soft_delete(self, source: KnowledgeSourcePO) -> KnowledgeSourcePO:
        source.status = "deleted"
        await self._session.flush()
        await self._session.refresh(source)
        return source

def sanitize_page_context(value: dict[str, Any] | None) -> dict[str, str]:
    if not isinstance(value, dict):

    sanitized: dict[str, str] = {}
    path = value.get("path")

  return value

  const [selectedId, setSelectedId] = useState<string | null>(null);

    try {
      const response = await fetchKnowledgeSources();
      setItems(response.items);
      setError(null);
    } catch (loadError) {
      setError(loadError instanceof Error ? loadError.message : "加载知识库失败");
    }

    void load();
  }, []);

    setForm({
      name: item.name,
      source_kind: sourceKindForForm(item.source_kind),

import type { AdvisorRecentMessage, AdvisorScene } from "@/types/ai";

export function advisorSessionStorageKey(scene: AdvisorScene, path: string) {
  return `advisor:${scene}:${path}`;
}

export function readAdvisorRecentMessages(scene: AdvisorScene, path: string): AdvisorRecentMessage[] {
  if (typeof window === "undefined") return [];
  const raw = window.sessionStorage.getItem(advisorSessionStorageKey(scene, path));
  if (!raw) return [];
  try {
    const parsed: unknown = JSON.parse(raw);
    if (!Array.isArray(parsed)) return [];
    return parsed
      .filter((item): item is AdvisorRecentMessage => item?.role === "user" || item?.role === "assistant")
      .slice(-6);
  } catch {
    return [];
  }
}

export function writeAdvisorRecentMessages(scene: AdvisorScene, path: string, messages: AdvisorRecentMessage[]) {
  if (typeof window === "undefined") return;
  window.sessionStorage.setItem(advisorSessionStorageKey(scene, path), JSON.stringify(messages.slice(-6)));
}
```
